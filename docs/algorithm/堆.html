堆

堆

堆主要有两个重要操作：

  • 上浮
  • 下沉

通过这两个核心的操作，可以得到堆的其他操作。

#include <iostream>
#include <vector>
using namespace std;

template<typename Comparable>
class BinaryHeap{
public:
    explicit BinaryHeap(int capacity = 100)
        :array(capacity),currentSize(0){}

    explicit BinaryHeap(const vector<Comparable> &items)
        :array(items.size() + 10), currentSize(items.size())
    {
        for(int i=0;i<items.size();i++){
            array[i+1] = items[i];
        }
        buildHeap();
    }

    bool isEmpty() const{
        return currentSize == 0;
    }

    void insert(const Comparable &x){
        if(currentSize == array.size() - 1){
            array.resize(array.size() * 2);
        }
        int hole = ++currentSize;
        Comparable copy = x;
        array[0] = std::move(copy);
        for(;x < array[hole/2];hole/=2){
            array[hole] = std::move(array[hole/2]);
        }
        array[hole] = std::move(array[0]);
    }

    Comparable deleteMin(){
        Comparable ret = std::move(array[1]);
        array[1] = std::move(array[currentSize--]);
        percolateDown(1);
        return ret;
    }

private:
    int currentSize;
    vector<Comparable> array;

    void buildHeap(){
        for(int i=currentSize/2;i>0;i--){
            percolateDown(i);
        }
    }

    void percolateDown(int hole){
        int child;
        Comparable tmp = std::move(array[hole]);
        for(;hole*2 <= currentSize; hole = child){
            child = hole * 2;
            if(child < currentSize && array[child + 1] < array[child]){
                ++child;
            }
            if(array[child] < tmp){
                array[hole] = std::move(array[child]);
            }else{
                break;
            }
        }
        array[hole] = std::move(tmp);
    }
};

#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

const int N = 1e5 + 10;
int hSize;
int h[N];
int kth[N]; //记录第i个插入元素在堆中的位置
int ord[N]; //记录堆的元素是第几个插入的

int idx;

void swapNode(int x, int y){
    swap(kth[ord[x]],kth[ord[y]]);
    swap(ord[x],ord[y]);
    swap(h[x],h[y]);
}

void up(int hole){
    // printf("up %d\n",hole);
    int pos = hole;
    while(pos > 1){
        int p = pos / 2;
        if(h[pos] < h[p]){
            swapNode(pos,p);
            pos = p;
        }else{
            break;
        }
    }
}

void down(int hole){
    int child;
    for(;hole*2 <= hSize; hole = child){
        child = hole*2;
        if(child < hSize && h[child + 1] < h[child]){
            child++;
        }
        if(h[hole] > h[child]){
            swapNode(hole,child);
        }else{
            break;
        }
    }
}

void insert(int x){
    h[++hSize] = x;
    kth[++idx] = hSize;
    ord[hSize] = idx;
    up(hSize);
}

void printHeap(){
    printf("heap info:");
    for(int i=1;i<=hSize;i++){
        printf("%d ",h[i]);
    }
    printf("\n");
}

int main(){
    int n;
    scanf("%d",&n);
    string op;
    int n1,n2;
    for(int i=0;i<n;i++){
        cin>>op;
        if(op == "I"){
            cin>>n1;
            insert(n1);
        }else if(op == "PM"){
            printf("%d\n",h[1]);
        }else if(op == "DM"){
            swapNode(1,hSize);
            hSize--;
            down(1);
        }else if(op == "D"){
            cin>>n1;
            int pos = kth[n1];
            swapNode(pos,hSize);
            hSize--;
            down(pos);
            up(pos);//这里是有可能需要执行up操作的，因为最后一个不一定大于该数
        }else if(op == "C"){
            cin>>n1>>n2;
            int pos = kth[n1];
            h[pos] = n2;
            up(pos);
            down(pos);
        }
    }
    return 0;
}
