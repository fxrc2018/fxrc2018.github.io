哈希表

哈希表

在我看来，哈希表实际上是对数组的扩充。在数组中，我们可以用下标对元素进行访问，时间为O(1)O(1)O(1)。但数组的缺陷在于，只支持数字作为索引，且元素最好为连续的，元素间隔太大时空间浪费很大。那么，怎么对数组进行优化呢？首先，要支持各种不同的元素，需要元素提供一个数字作为该元素的标识。其次，要将这些数字映射到一段连续的区间。最后，因为元素提供的数字范围大于区间的范围，因此在区间内必然产生冲突，需要一种冲突的解决办法。哈希表就是利用了这种思想，实现了对不同元素的访问。

template<typename HashObject, typename HashFunction>
class HashTable{
public:
    HashTable(int size = 101):
        currentSize(0),
        hashFunc(HashFunction()),
        theLists(size,list<HashObject>()){}

    void clear(){
        for(int i=0;i<theLists.size();i++){
            theLists[i].clear();
        }
        currentSize = 0;
    }

    bool contain(const HashObject &obj){
        list<HashObject> &whichList = theLists[myHash(obj)];
        return find(whichList.begin(),whichList.end(),obj) != whichList.end();
    }

    bool insert(const HashObject &obj){
        list<HashObject> &whichList = theLists[myHash(obj)];
        if(find(whichList.begin(),whichList.end(),obj) != whichList.end()){
            return false; // 已经有了
        }
        whichList.push_back(obj);
        currentSize++;
        if(currentSize > theLists.size()){
            rehash();
        }
        return true;
    }

    bool remove(const HashObject &obj){
        list<HashObject> &whichList = theLists[myHash(obj)];
        auto it = find(whichList.begin(),whichList.end(),obj);
        if(it == whichList.end()){
            return false; // 不存在
        }
        whichList.erase(it);
        currentSize--;
        return true;
    }

    int size(){
        return currentSize;
    }

    void printAll(){
        for(auto item : theLists){
            for(auto it = item.begin();it != item.end();++it){
                cout<<*it<<endl;
            }
        }
    }

private:
    int currentSize;
    HashFunction hashFunc;
    vector<list<HashObject>> theLists;

    int myHash(const HashObject &obj){
        int hashVal = hashFunc(obj);
        hashVal %= theLists.size();
        if(hashVal < 0){
            hashVal += theLists.size();
        }
        return hashVal;
    }

    void rehash(){
        vector<list<HashObject>> oldLists = theLists;
        theLists.resize(2 * theLists.size());
        clear();

        //拷贝原来的值
        for(int i=0;i<oldLists.size();i++){
            for(auto it = oldLists[i].begin();it != oldLists[i].end();++it){
                insert(*it);
            }
        }
    }
};

struct StringHash{
    int operator()(const string& key){
        int hashVal = 0;
        for(int i=0;i<key.size();i++){
            hashVal = 37 * hashVal + key[i]; // 看出一个37进制数
        }
        return hashVal;
    }
};


int main(int argc, char const *argv[])
{
    HashTable<string,StringHash> hashTable;
    for(int i = 0;i<200;i++){
        string str(i,'a');
        hashTable.insert(str);
        cout<<hashTable.size()<<endl;
    }
    // hashTable.printAll();
    system("pause");
    return 0;
}

#include <iostream>
using namespace std;

const int N = 1e5 + 10;
char str[N];
unsigned long long h[N];
unsigned long long p[N];

unsigned long long find(int l,int r){
    return h[r] - h[l - 1] * p[r-l+1]; 
}

int main(){
    int n,m;
    scanf("%d%d%s",&n,&m,str+1);
    p[0] = 1;
    for(int i=1;i<=n;i++){
        h[i] = h[i-1] * 131 + str[i];
        p[i] = p[i-1] * 131; 
    }
    int l1,r1,l2,r2;
    for(int i=0;i<m;i++){
        scanf("%d%d%d%d",&l1,&r1,&l2,&r2);
        if(find(l1,r1) == find(l2,r2)){
            printf("Yes\n");
        }else{
            printf("No\n");
        }
    }
    return 0;
}
